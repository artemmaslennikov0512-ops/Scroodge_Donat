# Вставить этот блок в /etc/nginx/sites-available/scrooge-donat на VPS
# (если файла нет: nano /etc/nginx/sites-available/scrooge-donat и вставить всё ниже)

# Только scrooge-donat.ru
server {
    listen 80;
    server_name scrooge-donat.ru www.scrooge-donat.ru;

    location /.well-known/acme-challenge/ {
        root /var/www/certbot;
    }

    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl;
    server_name scrooge-donat.ru www.scrooge-donat.ru;

    ssl_certificate     /etc/letsencrypt/live/scrooge-donat.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/scrooge-donat.ru/privkey.pem;

    location / {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
